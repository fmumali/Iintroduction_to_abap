   MODULE exit_screen INPUT.

  CASE ok_code.

   WHEN 'CANCEL'.
           CLEAR ok_code.
              LEAVE TO SCREEN 0.
         WHEN 'BACK'.
         LEAVE PROGRAM.
         ENDCASE.

  ENDMODULE.
